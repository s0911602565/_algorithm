    public static void sumTwo() {
         
        int num[] = {2, 7, 11, 15  , 20 , 13 , 200};// 15+ 13 = 28 ，回傳 5、6
        int target = 215;

        Map<Integer, Integer> map = new HashMap<>();
        for (int i = 0; i < num.length; i++) {
            int need = target - num[i];
            if(map.containsKey(need)){
                System.out.println(map.get(need) + " " + i);
            }
            map.put(num[i], i);
        }

    }
